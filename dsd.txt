                    type: 'POST', // Make sure you're using POST method
                    dataType: 'json',
                    success: function (data) {
                        if (data.success) {
                            // Handle success (e.g., show success message, redirect, etc.)
                            alert('Success: ' + data.message);
                        } else {
                            // Show error message in a popup (alert)
                            alert('Error: ' + data.message); // Display the error message from the JSON response
                        }
                    },
                    error: function (xhr, status, error) {
                        // Optionally handle failure
                        alert("An error occurred while processing your request.");
                        console.log("Error details: " + error);  // Log error details to console for debugging
                    }
                });
            });
